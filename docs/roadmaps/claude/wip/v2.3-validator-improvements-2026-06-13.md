5 problemas identificados no `trackfw validate` atual:

| # | Tipo | Problema |
|---|------|---------|
| B1 | Bug | `adr_dirs` não-recursivo → checks de ADR viram no-ops silenciosos com subpastas |
| B2 | Bug | Stale WIP por `mtime` → instável em CI (clone zera o mtime) |

| M4 | Melhoria | Sem coerência pasta × `status:` declarado no frontmatter |
| M5 | Melhoria | Mesmo filename em dois estados passa sem detecção |

**Mudança B2 — stale por git log:**
Adicionar função `gitLastModifiedTime(path string) (time.Time, bool)`:
```go
import "os/exec"
func gitLastModifiedTime(path string) (time.Time, bool) {
    cmd := exec.Command("git", "log", "-1", "--format=%ct", "--", path)
    out, err := cmd.Output()
    if err != nil || strings.TrimSpace(string(out)) == "" { return time.Time{}, false }
    ts, err := strconv.ParseInt(strings.TrimSpace(string(out)), 10, 64)
    if err != nil { return time.Time{}, false }
    return time.Unix(ts, 0), true
}
```
Em `validateStaleWIP()`: trocar `info.ModTime()` por:
```go
modTime, ok := gitLastModifiedTime(path)
if !ok { modTime = info.ModTime() }
```

**Mudança M3 — verificar existência de refs:**

**Mudança M4 — coerência pasta × status:**

### ML-1B — Node.js (`npm/src/validator/index.js`)
**Status:** ⬜ Pendente

**Arquivos afetados:**
- `npm/src/validator/index.js`

**Mudança B1 — adr_dirs recursivo:**
Adicionar função `walkDirMd(dir)` que retorna basenames de `.md` recursivamente:
```js
function walkDirMd(dir) {
  const results = []
  function walk(d) {
    let entries
    try { entries = fs.readdirSync(d) } catch (_) { return }
    for (const name of entries) {
      const full = path.join(d, name)
      try {
        if (fs.statSync(full).isDirectory()) { walk(full) }
        else if (name.endsWith('.md')) { results.push(name) }
      } catch (_) {}
    }
  }
  walk(dir)
  return results
}
```
Substituir `listDir(adrDir)` por `walkDirMd(adrDir)` em `validateADRsAreReferenced` e `validateFrontmatterPresence` (para ADRs).
Em `adrIsDraft(basename)`: substituir `path.join(adrDir, basename)` por busca recursiva:
```js
function findAdrFile(basename) {
  const cfg = config.load()
  for (const adrDir of cfg.adrDirs) {
    function search(d) {
      try {
        for (const name of fs.readdirSync(d)) {
          const full = path.join(d, name)
          try {
            if (fs.statSync(full).isDirectory()) { const r = search(full); if (r) return r }
            else if (name === basename) return full
          } catch (_) {}
        }
      } catch (_) {}
      return null
    }
    const found = search(adrDir)
    if (found) return found
  }
  return null
}
```

def _walk_dir_md(dir_path: str) -> list:

def _git_last_modified_time(file_path: str):
